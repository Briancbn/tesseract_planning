69 template <typename FloatType>
70 std::pair<bool, FloatType>
71 DescartesCollisionEdgeEvaluator<FloatType>::evaluate(const descartes_light::State<FloatType>& start,
72  const descartes_light::State<FloatType>& end) const
73 {
74  assert(start.values.rows() == end.values.rows());
75 
76  // Happens in two phases:
77  // 1. Compute the transform of all objects
78  tesseract_common::TrajArray segment(2, start.values.rows());
79  for (Eigen::Index i = 0; i < start.values.rows(); ++i)
80  {
81  segment(0, i) = start[i];
82  segment(1, i) = end[i];
83  }
84 
85  std::vector<tesseract_collision::ContactResultMap> discrete_results;
86  std::vector<tesseract_collision::ContactResultMap> continuous_results;
87  bool discrete_in_contact = discreteCollisionCheck(discrete_results, segment, allow_collision_);
88  bool continuous_in_contact = continuousCollisionCheck(continuous_results, segment, allow_collision_);
89 
90  if (!discrete_in_contact && !continuous_in_contact)
91  return std::make_pair(true, 0);
92 
93  // TODO: Update this to consider link pairs
94  auto collision_safety_margin_ =
95  static_cast<FloatType>(collision_check_config_.collision_margin_data.getMaxCollisionMargin());
96 
97  if (!discrete_in_contact && continuous_in_contact && allow_collision_)
98  return std::make_pair(true, collision_safety_margin_ - continuous_results.begin()->begin()->second[0].distance);
99 
100  if (discrete_in_contact && !continuous_in_contact && allow_collision_)
101  return std::make_pair(true, collision_safety_margin_ - discrete_results.begin()->begin()->second[0].distance);
102 
103  if (discrete_in_contact && continuous_in_contact && allow_collision_)
104  {
105  double d = collision_safety_margin_ - discrete_results.begin()->begin()->second[0].distance;
106  double c = collision_safety_margin_ - continuous_results.begin()->begin()->second[0].distance;
107  return std::make_pair(true, std::max(d, c));
108  }
109 
110  return std::make_pair(false, 0);
111 }

119 template <typename FloatType>
120 bool DescartesCollisionEdgeEvaluator<FloatType>::continuousCollisionCheck(
121  std::vector<tesseract_collision::ContactResultMap>& results,
122  const tesseract_common::TrajArray& segment,
123  bool find_best) const
124 {
125  // It was time using chronos time elapsed and it was faster to cache the contact manager
126  unsigned long int hash = std::hash<std::thread::id>{}(std::this_thread::get_id());
127  tesseract_collision::ContinuousContactManager::Ptr cm;
128  mutex_.lock();
129  auto it = continuous_contact_managers_.find(hash);
130  if (it == continuous_contact_managers_.end())
131  {
132  cm = continuous_contact_manager_->clone();
133  continuous_contact_managers_[hash] = cm;
134  }
135  else
136  {
137  cm = it->second;
138  }
139  mutex_.unlock();
140  tesseract_collision::CollisionCheckConfig config = collision_check_config_;
141  if (config.type == tesseract_collision::CollisionEvaluatorType::LVS_DISCRETE ||
142  config.type == tesseract_collision::CollisionEvaluatorType::LVS_CONTINUOUS)
143  config.type = tesseract_collision::CollisionEvaluatorType::LVS_CONTINUOUS;
144  else
145  config.type = tesseract_collision::CollisionEvaluatorType::CONTINUOUS;
146  config.contact_request.type =
147  (find_best) ? tesseract_collision::ContactTestType::CLOSEST : tesseract_collision::ContactTestType::FIRST;
148 
149  return tesseract_environment::checkTrajectory(results, *cm, *manip_, segment, config);
150 }
